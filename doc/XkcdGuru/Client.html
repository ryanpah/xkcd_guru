def comic_by_date(year, month = nil, day = nil)
  if day
    day_text = "&day=#{day}"
  end
  if month
    month_text = "&month=#{month}"
   end
  return HTTParty.get("#{@url}?year=#{year}#{month_text}#{day_text}")
 
end
